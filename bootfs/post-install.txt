#!/bin/bash

echo "post-install - Creating upstart job for reboot"
	(
	    echo description     "FirePick Packages Installation" 
	    echo start on startup
	    echo task
	    echo exec /boot/FirePick_Install.sh
	) >> /rootfs/etc/init/firepick_install.conf

	echo "bcm2709-rng" >> /etc/modules

#EOF
